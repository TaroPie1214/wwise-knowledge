# 为游戏对象设置多个位置

|  |
| --- |
| Wwise SDK 2025.1.4 - Windows |

为游戏对象设置多个位置

按照 "positions" 中的定义为 "gameObject" 设置多个位置以模拟一个在播声音对应的多个声源。

# 函数 URI

[ak.soundengine.setMultiplePositions](ak_soundengine_setmultiplepositions.html)

# 参数

{

"gameObject": 1122334,

"positions": [

{

"position": {

"orientationFront": {

"x": 5,

"y": 20,

"z": 10

},

"orientationTop": {

"x": 70,

"y": 150,

"z": 6

},

"position": {

"x": 70,

"y": 150,

"z": 10

}

}

},

{

"position": {

"orientationFront": {

"x": -50,

"y": 80,

"z": 15

},

"orientationTop": {

"x": 2,

"y": 40,

"z": 6

},

"position": {

"x": 2,

"y": -20,

"z": 400

}

}

}

],

"multiPositionType": 1

}

# 结果

{}
